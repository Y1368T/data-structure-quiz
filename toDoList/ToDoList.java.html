package toDoList;

public class ToDoList {
    private Node head;

    public ToDoList() {
        this.head = null;
    }

    public void addTodo(Task task) {
        Node newNode = new Node(task);
        if (head == null) {
            head = newNode;
        } else {
            Node temp = head;
            while (temp.next != null) {
                temp = temp.next;
            }
            temp.next = newNode;
        }
    }

    public void markTodoAsCompleted(String title) {
        Node temp = head;
        while (temp != null) {
            if (temp.task.getTitle().equals(title)) {
                temp.task.markCompleted();
                return;
            }
            temp = temp.next;
        }
    }

    public void viewTodoList() {
        Node temp = head;
        while (temp != null) {
            System.out.println("Title: " + temp.task.getTitle());
            System.out.println("Description: " + temp.task.getDescription());
            System.out.println("Completed: " + temp.task.isCompleted());
            System.out.println();
            temp = temp.next;
        }
    }
}
