package stack;
public class Stack {

    Node top;
    int length;

    public static class Node {
        int data;
        Node next;

        public Node(int data) {
            this.data = data;
            this.next = null;
        }
    }

    public Stack() {
        top = null;
        length = 0;
    }

    public int length() {
        return length;
    }

    public boolean isEmpty() {
        return length == 0;
    }

    public void push(int data) {
        Node temp = new Node(data);
        temp.next = top;
        top = temp;
        length++;
    }

    public int pop() {
        if (isEmpty()){
            System.out.println("Stack is empty");
        }
        int result = top.data;
        top = top.next;
        length--;
        return result;
    }

    public void peek() {
        if (isEmpty()) {
            System.out.println("Stack is empty");
        }
        System.out.println("the value:" + top.data);
    }

    public static void main(String[] args) {
        Stack stack = new Stack();
        stack.push(10);
        stack.push(15);
        stack.push(20);

        stack.peek();
        stack.pop();
        stack.peek();
        stack.pop();
        stack.peek();
    }
}
